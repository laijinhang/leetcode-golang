[5625. 比赛中的配对次数](https://leetcode-cn.com/problems/count-of-matches-in-tournament/)
```golang
func numberOfMatches(n int) int {
    if n == 0 || n == 1 {
        return 0
    }
    if n % 2 == 0 {
        return n / 2 + numberOfMatches(n / 2)
    }
    return (n-1) / 2 + numberOfMatches((n-1)/2+1)
}
```
