[96. 不同的二叉搜索树](https://leetcode-cn.com/problems/unique-binary-search-trees/)

```go
func numTrees(n int) int {
    dp := make([]int, n + 1)
    dp[0], dp[1] = 1, 1
    for i := 2;i <= n;i++ {
        for j := 0;j < i;j++ {
            dp[i] += dp[j] * dp[i-j-1]
        }
    }
    return dp[n]
}
```
